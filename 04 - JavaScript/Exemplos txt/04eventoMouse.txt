    <script>
        // O evento só funciona dentrode uma função
        function mensagem(){
            window.alert("Você clicou em um botão");
        }

        function enviar(){
            // obter o elemento input
            let nome = document.getElementById("nome");

            // condicional
            if(nome.value == ""){
                alert("Informe um nome, por favor");

            } else if (nome.value.length < 3){
                alert("O nome precisa ter pelo menos três caracteres");
                
            }else{
                alert("Nome informado: " + nome.value);
            }

            // Limpar elemento input
            nome.value = "";
            // Cursor dentro do elemento input
            nome.focus();
        }
        
    </script>
</head>
<body>

    <button onclick="mensagem()">Clique Aqui!</button>

    <hr>

    <input type="text" id="nome" placeholder="Informe um nome">
    <button onclick="enviar()">Enviar</button>
